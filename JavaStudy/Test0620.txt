/* Test062

다중for문을 활용한 별찍기 실습

다음과 같은 내용이 출력될수 있도록 반복문의 중첩 구문을 작성하여 프로그램구현

     *    1
    ***   3
   *****   5
  *******  7
 ********* 9

*/



public class Test062
{

	public static void main(String[] args)
		
	{
		int i,j,z; //i는 열 j는 빈칸 z는 *

		
		for(i=1;i<=5;i++)
		{

			for(j=4;j>=i;j--) //단순히 10번을 실행해도 되지만 그러면 j==5와 같은 조건을 붙여야할듯 //j는 빈칸이다.
						      //별을 찍는다보다 공백을 찍고 별을 찍는다는 생각으로 첫줄부터 공백 4 ->3 ->2 -> 1->0 만큼 찍는다.
			                   
			{
				System.out.print(" "); //처음에이걸만들고 *로 찍어보자 위의 공백개수만큼 찍힘



			}
			for(z=1;z<=2*i-1;z++){  //별의 개수가 1->9 까지 2씩 증가한다.
				System.out.print("*");

			}


			System.out.println("");
		}



	}


}
